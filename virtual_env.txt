# Create virtual env:
python -m venv <environment_name>
e.g.: python -m venv .git_profile_virtual_env

# Activate virtual env:
On Windows:
<environment_name>\Scripts\activate
e.g.: .git_profile_virtual_env\Scripts\activate

On macOS/Linux:
source <environment_name>/bin/activate
e.g.: source .git_profile_virtual_env/bin/activate

# Deactivate vitual env:
deactivate

# Delete virtual env:
On Windows:
rmdir /s /q <environment_name>
e.g.: rmdir /s /q .git_profile_virtual_env

Remove-Item -Recurse -Force <envirment_path>
e.g.: Remove-Item -Recurse -Force .\.git_profile_virtual_env

On macOS/Linux:
rm -rf <environment_name>
e.g.: source rm -rf .git_profile_virtual_env


# Install required packages
pip install -r requirements.txt
